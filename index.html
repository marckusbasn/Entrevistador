    let chat = null;

    let isLoading = false;

    function addMessage(text, isUser = false) {

        const messageDiv = document.createElement('div');

        messageDiv.classList.add('message');

        messageDiv.classList.add(isUser ? 'user-message' : 'gemini-message');

        messageDiv.innerText = text;

        chatBox.appendChild(messageDiv);

        chatBox.scrollTop = chatBox.scrollHeight;

    }

    async function sendMessage() {

        if (isLoading) return;

        const userMessage = chatInput.value.trim();

        if (userMessage === '') return;

        isLoading = true;

        sendButton.disabled = true;

        chatInput.value = '';

        addMessage(userMessage, true);

        try {

            const result = await chat.sendMessage(userMessage);

            const geminiResponse = result.response.text();

            addMessage(geminiResponse, false);

        } catch (error) {

            console.error("Erro ao enviar a mensagem:", error);

            addMessage("Desculpe, algo deu errado. Por favor, tente novamente.");

        } finally {

            isLoading = false;

            sendButton.disabled = false;

            chatInput.focus();

        }

    }

            },

        });

        const result = await chat.sendMessage("Olá!");

        const firstQuestion = result.response.text();

        addMessage(firstQuestion, false);

    }
